Inicializar webGL

var gl;

function initGL(canvas) {

try {

gl = canvas.getContext("ejemplo-canvas");

gl.viewport(0, 0, canvas.width, canvas.height);

} catch(e) {

}

if (!gl) {

alert("Could not initialise WebGL, sorry :-(");

}
